# Assignment 1
# Use .apply() to build a new feature with the counts for each of the selected_words
selected_words = ['awesome', 'great', 'fantastic', 'amazing', 'love', 'horrible', 'bad', 'terrible', 'awful', 'wow', 'hate']

def item_count(x):
    if item in x:             #if item is in dict x 
        count = x.get(item)   # function get() would not throw error
    else:
        count = 0
    return count

for item in selected_words:
    products[item] = products['word_count'].apply(item_count)   
    #every element in products['word_count'] is a dict
    # so fn apply() got a dict and transfer it to fn item_count as x
    print item, sum(products[item])
